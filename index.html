Q5.Take a variable in outer function and create an inner function to increase the counter every time it is called

        function outerFunc(num){
            console.log(`Value before inner function is called is : ${num}`);
            function innerFunc(){
                num++
                console.log(`Value after inner function is called is: ${num}`);
            }
            innerFunc();
        }
        outerFunc(1);
    
        OUTPUT:
        Value before inner function is called is :1
        Value after inner function is called is:2
